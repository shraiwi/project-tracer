345 bool tracer_parse_ble_payload(tracer_ble_payload payload, tracer_datapair * datapair) {
346 
347  bool output_valid = false, payload_valid = false;
348  for (uint8_t i = 0; i < payload.len;) {
349  uint8_t record_len = payload.value[i++];
350 
351  if (i + record_len > payload.len) return false; // segfault bad. >:( this protects the parser from them.
352 
353  uint8_t type = payload.value[i++];
354 
355  uint8_t * data = payload.value + i;
356  uint8_t data_len = record_len - 1;
357 
358  switch (type) {
359  case 0x03: // service uuid
360  payload_valid = *(uint16_t *)(data) == 0xFD6F; // check if the service id matches the contact tracing standard
361  break;
362  case 0x16: // service data
363  if (data_len == 2 + sizeof(datapair->rpi.value) + sizeof(datapair->aem.value)) { // check if the package is the right size
364  if (datapair) {
365  memcpy(datapair->rpi.value, data + 2, sizeof(datapair->rpi.value));
366  memcpy(datapair->aem.value, data + 2 + sizeof(datapair->rpi.value), sizeof(datapair->aem.value));
367  }
368  output_valid = true;
369  }
370  break;
371  }
372  i += data_len;
373  }
374 
375  return output_valid && payload_valid;
376 }
